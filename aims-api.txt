MODULE: AUTHENTICATION

[POST] /api/auth/register
Access: Admin Only
Auth: Public
Description: Register a new admin or user account
Request:
- first_name (string, required)
- last_name (string, required)
- email (string, required)
- password (string, required)
- password_confirm (string, required)
- role (string, required) [admin, manager, employee]
Response:
- id
- email
- role
- access_token
- refresh_token

[POST] /api/auth/login
Access: Mixed
Auth: Public
Description: Authenticate user and return JWT tokens
Request:
- email (string, required)
- password (string, required)
Response:
- access_token
- refresh_token
- user.id
- user.email
- user.role
- user.full_name
- user.avatar

[POST] /api/auth/token/refresh
Access: Mixed
Auth: Public
Description: Refresh access token using refresh token
Request:
- refresh (string, required)
Response:
- access
- refresh

[POST] /api/auth/logout
Access: Mixed
Auth: JWT
Description: Blacklist the refresh token to invalidate session
Request:
- refresh (string, required)
Response:
- detail

[POST] /api/auth/forgot-password
Access: Mixed
Auth: Public
Description: Send password reset link to user email
Request:
- email (string, required)
Response:
- detail

[POST] /api/auth/forgot-password/verify
Access: Mixed
Auth: Public
Description: Verify the password reset token is valid
Request:
- token (string, required)
- uid (string, required)
Response:
- valid (boolean)

[POST] /api/auth/reset-password
Access: Mixed
Auth: Public
Description: Reset user password using token and uid
Request:
- uid (string, required)
- token (string, required)
- new_password (string, required)
- new_password_confirm (string, required)
Response:
- detail

[POST] /api/auth/change-password
Access: Mixed
Auth: JWT
Description: Change password for currently authenticated user
Request:
- old_password (string, required)
- new_password (string, required)
- new_password_confirm (string, required)
Response:
- detail

[GET] /api/auth/me
Access: Mixed
Auth: JWT

Request: none

MODULE: TWO-FACTOR AUTHENTICATION
----------------------------------------------------------------

[POST] /api/auth/2fa/enable
Access: Mixed
Auth: JWT
Description: Initialize 2FA setup and return QR code URI
Request: none
Response:
- otpauth_url
- secret
- qr_code_base64

[POST] /api/auth/2fa/confirm
Access: Mixed
Auth: JWT
Description: Confirm and activate 2FA using TOTP code
Request:
- code (string, required)
Response:
- detail
- backup_codes[]

[POST] /api/auth/2fa/disable
Access: Mixed
Auth: JWT
Description: Disable 2FA for the authenticated user
Request:
- code (string, required)
Response:
- detail

[POST] /api/auth/2fa/verify
Access: Mixed
Auth: Public
Description: Verify TOTP code after initial login when 2FA is enabled
Request:
- temp_token (string, required)
- code (string, required)
Response:
- access_token
- refresh_token

[POST] /api/auth/2fa/backup
Access: Mixed
Auth: Public
Description: Use a backup code to authenticate when TOTP device is unavailable
Request:
- backup_code (string, required)
Response:
- access_token
- refresh_token

[GET] /api/auth/2fa/backup-codes
Access: Mixed
Auth: JWT
Description: List remaining valid backup codes (masked)
Request: none
Response:
- remaining_count
- codes[]

[POST] /api/auth/2fa/backup-codes/regenerate
Access: Mixed
Auth: JWT
Description: Regenerate a fresh set of backup codes, invalidating old ones

- code (string, required)
Response:
- backup_codes[]
